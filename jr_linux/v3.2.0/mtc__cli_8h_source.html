144 do \
145 { \
146 /* get current profile user */ \
147 ZCHAR *pcCurUser = Mtc_ProvDbGetCurProfUser(); \
148 /* check the profile is exist, then open client if it is exist*/ \
149 if (!ZOS_ISEMPTY_STR(pcCurUser) && Mtc_ProfExistUser(pcCurUser)) \
150 _ret = Mtc_CliOpen(pcCurUser); \
151 else \
152 _ret = ZFAILED; \
153 } while (0)

164 } \
